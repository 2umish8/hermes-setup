```bash
#!/usr/bin/env bash
# monitor-cpu.sh – run via cron every minute
set -euo pipefail

# ------------------------------------------------------------------
# 1. Measure total CPU usage (sum of all non‑idle percentages)
# ------------------------------------------------------------------
cpu_usage=$(top -bn1 | awk '
/Cpu/ {
    sum = 0
    for (i = 2; i <= NF; i++) {
        gsub(/[^0-9.]/, "", $i)   # strip commas, percent signs, text
        sum += $i
    }
    printf "%.1f", sum
}')

# If usage is >300%, proceed
if (( $(echo "$cpu_usage > 300" | bc -l) )); then
    # ------------------------------------------------------------------
    # 2. Wait 2 minutes and re‑check
    # ------------------------------------------------------------------
    sleep 120

    cpu_usage2=$(top -bn1 | awk '
    /Cpu/ {
        sum = 0
        for (i = 2; i <= NF; i++) {
            gsub(/[^0-9.]/, "", $i)
            sum += $i
        }
        printf "%.1f", sum
    }')

    if (( $(echo "$cpu_usage2 > 300" | bc -l) )); then
        # ------------------------------------------------------------------
        # 3. Gather data for the log
        # ------------------------------------------------------------------
        LOGDIR="/home/hermes/.hermes/logs"
        mkdir -p "$LOGDIR"
        TS=$(date +%Y%m%d%H%M%S)
        LOGFILE="${LOGDIR}/cpu-spike-${TS}.log"

        echo "=== CPU Spike Detected ===" >  "$LOGFILE"
        echo "Timestamp: $(date)" >> "$LOGFILE"
        echo "Total CPU usage: ${cpu_usage2}%" >> "$LOGFILE"
        echo >> "$LOGFILE"

        # Top process causing the spike
        TOP_PROC=$(ps aux --sort=-%cpu -o pid,ppid,%cpu,cmd --no-header | head -n 1)
        echo "Top process (by %CPU):" >> "$LOGFILE"
        echo "$TOP_PROC" >> "$LOGFILE"
        echo >> "$LOGFILE"

        # Last 10 minutes of journalctl logs
        echo "=== journalctl (last 10 minutes) ===" >> "$LOGFILE"
        journalctl --since "10 minutes ago" >> "$LOGFILE"

        # ------------------------------------------------------------------
        # 4. Notify via stdout (cron will email this to the owner)
        # ------------------------------------------------------------------
        echo "CPU spike of ${cpu_usage2}% detected. Log saved to $LOGFILE"
    fi
fi
```
