#!/bin/bash

# Script to monitor CPU usage and log spikes.
# Designed to be run by a cron job once per minute.

# --- Configuration ---
# CPU threshold (total active CPU utilization across all cores).
# Exceeding this value triggers the spike detection and logging process.
# The value is the sum of percentages for %usr, %nice, %sys, %iowait, %irq, %soft across all cores.
CPU_THRESHOLD=300.0 # e.g., 300%

# Time to wait (in seconds) after the initial spike is detected before re-checking.
WAIT_TIME_SECONDS=120 # 2 minutes

# Directory to store log files.
LOG_DIR="/home/hermes/.hermes/logs"

# Command to get system logs for the last 10 minutes.
LOG_LINES_CMD="journalctl --since '10 minutes ago'"

# Command to identify top CPU-consuming processes.
# Includes header and the top 2 processes (sorted by %CPU).
TOP_PROCESS_CMD="ps aux --sort=-%cpu | head -n 2"

# --- Helper Function: Get Total CPU Load ---
# Calculates the sum of active CPU utilization across all cores.
# Returns 1 if the total load exceeds CPU_THRESHOLD, 0 otherwise.
# Returns 2 on critical error (e.g., mpstat failure).
get_total_cpu_load() {
    # Get CPU statistics per-CPU over a 1-second interval.
    # Suppress stderr in case mpstat is unavailable or fails.
    CPU_STATS_PER_CPU=$(mpstat -P ALL 1 1 2>/dev/null)

    if [ $? -ne 0 ] || [ -z "$CPU_STATS_PER_CPU" ]; then
        echo "Error: mpstat command failed or returned no output on this check." >&2
        # Log persistent mpstat errors to a dedicated file for debugging.
        echo "Error: mpstat command failed. $(date)" >> "$LOG_DIR/mpstat_failure.log"
        return 2 # Indicate a critical error
    fi

    TOTAL_CPU_LOAD=0
    # Process each line starting with a digit (CPU stats lines) from mpstat -P ALL output.
    # Summing %usr, %nice, %sys, %iowait, %irq, %soft (columns 3 through 7).
    # These columns represent primary active CPU usage.
    TOTAL_CPU_LOAD=$(echo "$CPU_STATS_PER_CPU" | awk '
    /^[0-9]/ { # Matches lines starting with a digit, representing individual CPUs
        cpu_active_sum = 0;
        # Summing columns 3 to 7 for active CPU metrics: %usr, %nice, %sys, %iowait, %irq, %soft
        for (i = 3; i <= 7; i++) {
            cpu_active_sum += $i;
        }
        # Accumulate the sum for this CPU line into a running total across all CPUs.
        # This sum represents the "total across all cores" usage in percentage points.
        total_load_across_all_cpus += cpu_active_sum;
    }
    END {
        # Print the final accumulated sum.
        print total_load_across_all_cpus;
    }')

    if [ $? -ne 0 ] || [ -z "$TOTAL_CPU_LOAD" ]; then
        echo "Error: awk processing failed for mpstat output." >&2
        # Log error and exit script, indicating critical failure
        return 2
    fi

    # Compare total load with the threshold using bc for floating-point precision.
    if command -v bc &> /dev/null; then
        # bc -l enables floating-point arithmetic
        IS_OVER_THRESHOLD=$(echo "$TOTAL_CPU_LOAD > $CPU_THRESHOLD" | bc -l)
    else
        echo "Warning: 'bc' command not found. Using integer comparison for CPU threshold, which might be less accurate." >&2
        # Fallback to integer arithmetic if bc is not available.
        # This might truncate floating point numbers, leading to inaccuracies.
        TOTAL_CPU_LOAD_INT=$(printf "%.0f" "$TOTAL_CPU_LOAD") # Convert to integer for comparison
        IS_OVER_THRESHOLD=$((TOTAL_CPU_LOAD_INT > CPU_THRESHOLD))
    fi
    echo "$IS_OVER_THRESHOLD" # Return 1 if over threshold, 0 otherwise
    return 0 # Success
}

# --- Main Script Execution ---

# Create log directory if it doesn't exist.
mkdir -p "$LOG_DIR"

# Check initial CPU load.
INITIAL_LOAD_OVER_THRESHOLD=$(get_total_cpu_load)

# Check the return code of get_total_cpu_load to handle critical errors.
if [ $? -eq 2 ]; then
    # Critical error during initial check (e.g., mpstat failed). Exit script.
    exit 1
fi

if [ "$INITIAL_LOAD_OVER_THRESHOLD" -eq 1 ]; then
    # Spike detected on first check.
    echo "CPU usage spike detected: Total active CPU utilization is > ${CPU_THRESHOLD}%."

    # 1. Wait for the specified time.
    echo "Waiting for $WAIT_TIME_SECONDS seconds..."
    sleep $WAIT_TIME_SECONDS

    # Re-check CPU usage after waiting.
    AFTER_WAIT_LOAD_OVER_THRESHOLD=$(get_total_cpu_load)

    if [ $? -eq 2 ]; then
        # Critical error during re-check. Exit script.
        exit 1
    fi

    if [ "$AFTER_WAIT_LOAD_OVER_THRESHOLD" -eq 1 ]; then
        # CPU usage is still high after waiting. Proceed to log.
        echo "CPU usage still above threshold (${CPU_THRESHOLD}%) after waiting. Logging details..."

        # Generate a timestamp for the log file name.
        TIMESTAMP=$(date +"%Y%m%d%H%M%S")
        LOG_FILE="${LOG_DIR}/cpu-spike-${TIMESTAMP}.log"

        echo "Logging CPU spike details to: $LOG_FILE"

        # Get top processes information. Redirect stderr to /dev/null in case ps fails.
        TOP_PROCESSES=$(eval "$TOP_PROCESS_CMD" 2>/dev/null || echo "<Error retrieving top processes>")

        # Execute journalctl, capture output, and handle potential errors.
        JOURNAL_OUTPUT=$($LOG_LINES_CMD 2>/dev/null || echo "<Error executing journalctl command or no logs found>")

        # Combine all information into a report and write to the log file using tee.
        # 'tee' writes to stdout (for system delivery) and the specified file.
        {
            echo "--- CPU Spike Detected ---"
            echo "Timestamp: $(date '+%Y-%m-%d %H:%M:%S')"
            echo "Initial Total Active CPU Utilization: > ${CPU_THRESHOLD}%"
            echo "CPU Utilization after ${WAIT_TIME_SECONDS}s wait: > ${CPU_THRESHOLD}%"
            echo "Number of CPU cores: $(nproc || echo '<Unknown>')" # Include num cores for context
            echo ""
            echo "--- Top Processes Causing Spike ---"
            echo "$TOP_PROCESSES"
            echo ""
            echo "--- Last 10 Minutes of System Logs (journalctl) ---"
            echo "$JOURNAL_OUTPUT"
            echo "--- End of Logs ---"
        } | tee "$LOG_FILE"

        # 3. Notify the user by printing to stdout, which the system will deliver.
        echo "Notification: High CPU usage detected and logged. Details saved to: $LOG_FILE"
    else
        # CPU usage dropped below threshold after waiting.
        echo "CPU usage dropped below threshold (${CPU_THRESHOLD}%) after the wait. No spike recorded."
    fi
else if [ "$INITIAL_LOAD_OVER_THRESHOLD" -eq 0 ]; then
    # CPU usage is within normal limits. No action needed.
    # Uncomment the following line if you want to see confirmation messages on normal checks.
    # echo "CPU usage is within normal limits. No spike detected."
    : # Do nothing.
else
    # An unexpected return code from get_total_cpu_load, indicating an error.
    echo "Script execution encountered an unexpected error state." >&2
    exit 1
fi

exit 0

*   It then calculates the "total across all cores" utilization by summing the active percentage columns (`%usr`, `%nice`, `%sys`, `%iowait`, `%irq`, `%soft`) for *each* core and then summing these per-core totals.
*   If this total utilization exceeds `300.0%`, it waits for 2 minutes.
*   After waiting, it checks the CPU usage again. If it's still above the threshold:
    *   It gets the last 10 minutes of `journalctl` logs.
    *   It identifies the top 2 CPU-consuming processes using `ps aux --sort=-%cpu`.
    *   It consolidates this information into a timestamped log file in `/home/hermes/.hermes/logs/`.
    *   It prints a notification message to standard output, which will be delivered by the system.

This setup ensures that the CPU usage is monitored continuously every minute, and detailed logs are generated only when a persistent spike occurs.
