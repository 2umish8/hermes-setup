```bash
#!/usr/bin/env bash
# monitor_cpu.sh – cron job to detect CPU spikes and dump recent logs

# 1. Compute current total CPU usage (% across all cores)
CPU_TOTAL=$(ps -eo pcpu --no-headers | awk '{sum+=$1} END{print sum}')

# Function to report a spike
report_spike() {
    local now="$(date +%Y-%m-%dT%H:%M:%S)"
    local ts="$(date +%Y%m%d%H%M%S)"
    local logfile="/home/hermes/.hermes/logs/cpu-spike-${ts}.log"
    mkdir -p "$(dirname "$logfile")"

    # Last 10 min of journal logs
    journalctl --since="10 minutes ago" > "$logfile"

    # Append top process info
    echo >> "$logfile"
    echo "===== Top process (by %CPU) =====" >> "$logfile"
    ps aux --sort=-%cpu | sed -n '2p' >> "$logfile"

    # Notify user via stdout (cron will email this)
    echo "🚨 CPU spike detected at ${now}"
    echo "Total CPU usage: $(printf '%.2f' "$CPU_TOTAL")%"
    echo "Top process (by %CPU): $(ps aux --sort=-%cpu | sed -n '2p' | awk '{print $11}')"
    echo "Recent logs stored at: $logfile"
}

# 2. Check if CPU > 300%
if awk -v cpu="$CPU_TOTAL" 'BEGIN{exit cpu>300?0:1}'; then
    # CPU high – wait 2 minutes
    sleep 120

    # Re‑check CPU after waiting
    CPU_TOTAL=$(ps -eo pcpu --no-headers | awk '{sum+=$1} END{print sum}')
    if awk -v cpu="$CPU_TOTAL" 'BEGIN{exit cpu>300?0:1}'; then
        report_spike
    fi
fi
```

**What it does**

1. Runs once per minute (cron).  
2. Computes the sum of all process %CPU values (total across all cores).  
3. If the sum exceeds 300 % → waits 2 min, re‑checks.  
4. If still > 300 % → dumps the last 10 min of `journalctl` output to a timestamped file under `/home/hermes/.hermes/logs/`, appends the top‑CPU‑using process, and prints a concise notification.  

No output is produced when there is no spike, so cron will not send an email (the task is silent).
