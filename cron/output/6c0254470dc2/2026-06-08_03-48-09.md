```bash
#!/usr/bin/env bash
set -euo pipefail

# Log directory
LOGDIR="/home/hermes/.hermes/logs"
mkdir -p "$LOGDIR"

# Gather total CPU usage (sum of all %cpu values)
cpu_total=$(ps -eo pcpu --no-headers | awk '{sum+=$1} END {print sum}')

# Check if total CPU exceeds 300%
if (( $(echo "$cpu_total > 300" | bc -l) )); then
    # Wait 2 minutes
    sleep 120

    # Re‑check
    cpu_total=$(ps -eo pcpu --no-headers | awk '{sum+=$1} END {print sum}')
    if (( $(echo "$cpu_total > 300" | bc -l) )); then
        # Prepare log file
        stamp=$(date +%Y%m%d%H%M%S)
        LOGFILE="$LOGDIR/cpu-spike-$stamp.log"

        # Write diagnostics
        {
            echo "CPU spike detected at $(date)"
            echo "Total CPU usage after 2 minutes: $cpu_total%"
            echo
            echo "Top process causing spike:"
            ps aux --sort=-%cpu | head -n 2
            echo
            echo "Full syslog for last 10 minutes:"
            journalctl --since "10 minutes ago"
        } > "$LOGFILE"

        # Notify (cron will email the output to the user if configured)
        echo "CPU spike >300%. Log saved to $LOGFILE."
    fi
fi
```
