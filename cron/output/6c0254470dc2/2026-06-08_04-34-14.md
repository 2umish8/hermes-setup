#!/usr/bin/env bash
# /home/hermes/.hermes/scripts/cpu_monitor.sh
# Run via cron every minute

set -euo pipefail

# Get current total CPU usage (percent across all cores)
cpu_pct() {
    # Using top in batch mode; parses "Cpu(s)" line
    # The line looks like: Cpu(s):  1.5%us,  0.5%sy, ...
    # We compute 100 - idle% (the last number) to get usage
    top -bn1 \
        | grep 'Cpu(s)' \
        | awk -F',' '{last=$NF; gsub(/[[:space:]]*%,/,"",last); printf "%.0f\n", 100 - last}'
}

check_cpu() {
    echo "$(cpu_pct)"
}

# --- main logic --------------------------------------------------

# First check
cpu1=$(check_cpu)
if (( $(echo "$cpu1 > 300" | bc -l) )); then
    # Wait 2 minutes before re‑checking
    sleep 120
    cpu2=$(check_cpu)
    if (( $(echo "$cpu2 > 300" | bc -l) )); then
        # 10‑minute journal log + top process
        timestamp=$(date +%Y%m%d%H%M%S)
        logfile="/home/hermes/.hermes/logs/cpu-spike-${timestamp}.log"
        mkdir -p "$(dirname "$logfile")"

        # Identify the top process causing the spike
        topproc=$(ps aux --sort=-%cpu | head -n 2 | tail -n 1)

        {
            echo "=== CPU spike detected ==="
            echo "Time: $(date -u '+%Y-%m-%d %H:%M:%S UTC')"
            echo "CPU usage: ${cpu2}% (threshold 300%)"
            echo
            echo "=== Top process ==="
            echo "$topproc"
            echo
            echo "=== Last 10 minutes of system logs (journalctl) ==="
            journalctl --since "10 minutes ago"
        } > "$logfile"

        # Simple notification – adjust to your own alerting mechanism
        echo "CPU spike detected (${cpu2}%). Log written to $logfile"
    fi
fi
