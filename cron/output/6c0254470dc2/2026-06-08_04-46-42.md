#!/usr/bin/env bash
# /home/hermes/.hermes/bin/check_cpu_spike.sh
#
# This script is meant to be run by a cron job every minute.
# If the total CPU usage across all cores exceeds 300% it waits 2 minutes,
# double‑checks, and if still above the threshold it dumps the last 10
# minutes of system logs to a timestamped file and prints a notification
# message (cron will mail or redirect this to the user).

set -euo pipefail

# ----- Helper -----------------------------------------------------------

# Calculate total CPU% across all processes
calc_total_cpu() {
    ps -e -o %cpu --no-headers | awk '{sum+=$1} END{print sum}'
}

# ----- Main logic -------------------------------------------------------

TOTAL_CPU=$(calc_total_cpu)

# Compare with 300% using bc for floating point
if (( $(echo "$TOTAL_CPU > 300" | bc -l) )); then
    # Wait 2 minutes before re‑checking
    sleep 120

    TOTAL_CPU_AFTER=$(calc_total_cpu)
    if (( $(echo "$TOTAL_CPU_AFTER > 300" | bc -l) )); then
        TIMESTAMP=$(date +%Y%m%d%H%M%S)
        LOG_FILE="/home/hermes/.hermes/logs/cpu-spike-${TIMESTAMP}.log"

        mkdir -p "$(dirname "$LOG_FILE")"

        {
            echo "===== CPU SPIKE DETECTED ====="
            echo "Time: $(date)"
            echo "Total CPU: ${TOTAL_CPU_AFTER}%"
            echo
            echo "Top process causing spike:"
            ps aux --sort=-%cpu | head -n 2 | tail -n 1
            echo
            echo "=== Last 10 minutes of journalctl ==="
            journalctl --since "10 minutes ago"
        } >"$LOG_FILE"

        # Notify the user – cron will capture this output in its mail
        echo "CPU spike detected at $(date). Logs written to $LOG_FILE"
    fi
fi
