#!/bin/bash

# Script Name: monitor_cpu_spike.sh
# Description: Monitors CPU usage and logs details when a spike is detected.
# Author: Gemini Agent
# Date: 2024-06-08

# --- Configuration ---
# CPU usage threshold: If the *total* CPU utilization across all cores
# exceeds this percentage, a spike is considered.
# Example: 300 means 300% utilization, equivalent to 3 full cores being utilized
# on a multi-core system.
CPU_THRESHOLD=300

# Number of minutes to wait after initial detection.
# A second check is performed after this delay to confirm the spike.
WAIT_MINUTES=2

# Directory where log files will be stored.
# Ensure this directory is writable by the user running the script.
LOG_DIR="/home/hermes/.hermes/logs"

# Number of minutes of journalctl logs to capture during a confirmed spike.
# Logs will be filtered for critical, alert, and emergency messages.
JOURNAL_LOG_MINUTES=10

# --- Script Logic ---

# Ensure the log directory exists. Creates it if it doesn't.
# The '-p' option ensures that parent directories are also created if needed,
# and it doesn't complain if the directory already exists.
mkdir -p "$LOG_DIR"

# Function to get the current total CPU usage percentage.
# It uses 'mpstat' to measure CPU activity over a short interval.
# Returns the usage percentage as a float, or "error" if it fails.
get_cpu_usage() {
    local cpu_usage_line
    # Run mpstat for 1 second, 1 time. Capture output.
    # Use awk to find the line starting with "Average:" and print its last field (the CPU percentage).
    # The $NF refers to the last field in the line.
    cpu_usage_line=$(mpstat 1 1 | awk '/^Average:/ {print $NF}')

    # Validate if the captured output is a number.
    # This regex checks for an optional integer part followed by an optional decimal part.
    if ! [[ "$cpu_usage_line" =~ ^[0-9]+(\.[0-9]+)?$ ]]; then
        echo "Error: Could not determine CPU usage. 'mpstat 1 1' output was unexpected: '$cpu_usage_line'" >&2
        echo "error" # Return "error" to indicate a problem.
        return 1
    fi
    echo "$cpu_usage_line" # Return the valid CPU usage percentage.
}

# Main function to monitor CPU and act on spikes.
monitor_cpu_spike() {
    local initial_cpu_usage
    local cpu_usage_after_wait
    local timestamp
    local log_file
    local top_processes
    local notification_message

    # 1. Get initial CPU usage.
    initial_cpu_usage=$(get_cpu_usage)
    if [[ "$initial_cpu_usage" == "error" ]]; then
        echo "Failed to get initial CPU usage. Exiting." >&2
        return 1
    fi

    # 2. Check if CPU usage exceeds or meets the threshold.
    # We use 'bc -l' for floating-point arithmetic and comparisons.
    # '(( $(echo "expression" | bc -l) ))' evaluates to 1 (true) or 0 (false).
    if (( $(echo "$initial_cpu_usage >= $CPU_THRESHOLD" | bc -l) )); then
        # Initial detection of a potential spike.
        echo "CPU spike detected: ${initial_cpu_usage}% (Threshold: ${CPU_THRESHOLD}%)"

        # 3. Wait for the configured duration to confirm the spike.
        echo "Waiting ${WAIT_MINUTES} minute(s) for confirmation..."
        sleep $(($WAIT_MINUTES * 60)) # Sleep for duration in seconds.

        # 4. Check CPU usage again after waiting.
        cpu_usage_after_wait=$(get_cpu_usage)
        if [[ "$cpu_usage_after_wait" == "error" ]]; then
            echo "Warning: Failed to re-determine CPU usage after waiting. Logging based on initial detection." >&2
            # If re-check fails, we still log based on the initial detection if it was high.
            # This prevents missing a spike due to transient issues with reporting.
            cpu_usage_after_wait=$initial_cpu_usage
        fi

        # 5. Confirm spike and proceed with logging if still high.
        if (( $(echo "$cpu_usage_after_wait >= $CPU_THRESHOLD" | bc -l) )); then
            echo "CPU spike confirmed. Initiating logging..."

            # Generate a timestamp for the log file.
            timestamp=$(date +%Y%m%d%H%M%S)
            log_file="${LOG_DIR}/cpu-spike-${timestamp}.log"

            # Identify the top processes consuming CPU.
            # 'ps aux --sort=-%cpu' lists all processes sorted by CPU usage (descending).
            # 'head -n 11' gets the header line plus the top 10 processes.
            top_processes=$(ps aux --sort=-%cpu | head -n 11)

            # Create the log file and write details.
            # The redirection '>' overwrites the file if it exists (though with timestamp it shouldn't).
            # '>>' appends to the file.
            echo "--- CPU Spike Log ---" > "$log_file"
            # Get actual log file creation time for accuracy using stat.
            echo "Log File Creation Time: $(date -d@$(stat -c %Y "$log_file") '+%Y-%m-%d %H:%M:%S')" >> "$log_file"
            echo "Initial CPU Usage Detected: ${initial_cpu_usage}%" >> "$log_file"
            echo "CPU Usage After Wait: ${cpu_usage_after_wait}%" >> "$log_file"
            echo "" >> "$log_file" # Add a blank line for readability.
            echo "--- Top Processes (sorted by %CPU) ---" >> "$log_file"
            echo "$top_processes" >> "$log_file"
            echo "" >> "$log_file"
            echo "--- Journalctl Logs (last ${JOURNAL_LOG_MINUTES} minutes, critical/alert/emerg) ---" >> "$log_file"
            # Capture critical, alert, and emergency messages from journalctl.
            # '--since' argument specifies the time; '-p' filters by priority.
            journalctl --since "${JOURNAL_LOG_MINUTES} minutes ago" -p crit,alert,emerg >> "$log_file"
            echo "--- End of Log ---" >> "$log_file"

            # Prepare notification message for the user.
            # This message will be printed to standard output. If this script is run
            # via cron, this output is typically mailed to the user associated with cron.
            notification_message="CPU spike detected! Total CPU usage was ${cpu_usage_after_wait}%. Detailed logs saved to: ${log_file}"
            echo "$notification_message"

            return 0 # Spike handled successfully.
        else
            # Spike was not confirmed after waiting.
            echo "CPU usage dropped below threshold after waiting. Spike not confirmed."
            return 0 # Normal execution, no spike.
        fi
    else
        # No spike detected initially.
        # Uncomment the line below if you want status messages even when no spike occurs.
        # echo "CPU usage (${initial_cpu_usage}%) is within the normal range."
        return 0 # Normal execution, no spike.
    fi
}

# Execute the main monitoring function.
monitor_cpu_spike
