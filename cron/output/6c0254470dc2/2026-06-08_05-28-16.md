total_cpu() {

# Helper: get current top CPU process (full line from ps)

}

LOGDIR="/home/hermes/.hermes/logs"
mkdir -p "$LOGDIR"

CUR_CPU=$(total_cpu | awk '{print int($1+0.5)}')  # round to nearest int
if (( CUR_CPU > THRESHOLD )); then
    # 2. Wait 2 minutes
    sleep 120

    CUR_CPU=$(total_cpu | awk '{print int($1+0.5)}')

            echo "Total CPU: ${CUR_CPU}% (threshold ${THRESHOLD}%)"
            echo "Top process (CPU %):"
