```bash
#!/usr/bin/env bash
# /usr/local/bin/cpu_spike_monitor.sh
#
# Checks total CPU utilisation. If >300% triggers a two‑minute wait,
# re‑checks, and if still >300% writes the last 10 minutes of system logs
# (journalctl) plus the top offending process to a timestamped file.
# It then logs a notification message via syslog – cron will email it
# to the user “hermes”.

set -euo pipefail

# Absolute paths for safety
LOG_DIR="/home/hermes/.hermes/logs"
LOG_FILE="${LOG_DIR}/cpu-spike-$(date +%Y%m%d%H%M%S).log"
JOURNAL_CMD="/usr/bin/journalctl"
PS_CMD="/usr/bin/ps"
AWK_CMD="/usr/bin/awk"
DATE_CMD="/usr/bin/date"
LOGGER="/usr/bin/logger"

# Ensure log directory exists
mkdir -p "${LOG_DIR}"

# Calculate total CPU utilisation (sum of %CPU from ps)
# ps aux formatting: pid, user, %cpu, %mem, command, etc.
TOTAL_CPU=$("${PS_CMD}" -eo "%cpu" | "${AWK_CMD}" '{s += $1} END {print s}')

# Threshold check
THRESHOLD=300
if (( $(echo "$TOTAL_CPU > $THRESHOLD" | /usr/bin/bc -l) )); then
    # Wait 2 minutes
    sleep 120

    # Re‑check
    TOTAL_CPU_RECHECK=$("${PS_CMD}" -eo "%cpu" | "${AWK_CMD}" '{s += $1} END {print s}')
    if (( $(echo "$TOTAL_CPU_RECHECK > $THRESHOLD" | /usr/bin/bc -l) )); then
        # Grab the top process causing the spike
        TOP_PROC=$("${PS_CMD}" aux --sort=-%cpu | head -n 2 | tail -n 1)

        # Dump last 10 minutes of journalctl logs
        "${JOURNAL_CMD}" -b --since "10 minutes ago" > "${LOG_FILE}"

        # Append top offending process information
        echo "----- TOP PROCESS CAUSED CPU SPIKE -----" >> "${LOG_FILE}"
        echo "${TOP_PROC}" >> "${LOG_FILE}"
        echo "----- END -----" >> "${LOG_FILE}"

        # Notify the user (cron will email this message to hermes)
        "${LOGGER}" -p user.warning "CPU spike detected (>${THRESHOLD}%). Log written to ${LOG_FILE}"
    fi
fi
```
